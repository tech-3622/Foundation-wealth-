import foundation as fw
import pandas as pd
 
class MomentumStrategy(fw.Strategy):
def __init__(self):
super().__init__()
self.lookback = 14
self.threshold = 1.02
 
def on_data(self, data: pd.DataFrame):
# Calculate momentum indicator
momentum = data['close'].pct_change(14)
if momentum.iloc[-1] > 0.02:
self.buy(symbol='AAPL', qty=100)
elif momentum.iloc[-1] < -0.02:
self.sell(symbol='AAPL', qty=100)
 
def backtest(self):
results = self.run_backtest(
start='2024-01-01',
end='2024-12-31',
capital=100000
)
return results.sharpe_ratio
